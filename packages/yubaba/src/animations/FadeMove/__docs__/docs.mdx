# FadeMove

Will move the origin element to the destination element while fading out.

## Usage

<Playground>
  <Common.Toggler>
    {({ shown, toggle }) =>
      shown ? (
        <Baba name="fade-move" key="1">
          <FadeMove>
            {({ ref, style }) => (
              <Styled.Box onClick={() => toggle()} style={style} innerRef={ref}>
                Hello, world!
              </Styled.Box>
            )}
          </FadeMove>
        </Baba>
      ) : (
        <Baba name="fade-move" key="2">
          <FadeMove>
            {({ ref, style }) => (
              <Styled.Box style={style} innerRef={ref} onClick={() => toggle()} alignRight />
            )}
          </FadeMove>
        </Baba>
      )
    }
  </Common.Toggler>
</Playground>
